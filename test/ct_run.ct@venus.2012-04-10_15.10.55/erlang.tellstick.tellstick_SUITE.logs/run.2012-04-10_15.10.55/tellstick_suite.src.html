    9:-module(tellstick_SUITE).
   10:
   11:-compile(export_all).
   12:
   13:-include_lib("common_test/include/ct.hrl").
   14:
   15:%%--------------------------------------------------------------------
   16:%% @spec suite() -> Info
   17:%% Info = [tuple()]
   18:%% @end
   19:%%--------------------------------------------------------------------
   20:suite() ->
   21:    [{timetrap,{seconds,30}}].
   22:
   23:%%--------------------------------------------------------------------
   24:%% @spec all() -> GroupsAndTestCases | {skip,Reason}
   25:%% GroupsAndTestCases = [{group,GroupName} | TestCase]
   26:%% GroupName = atom()
   27:%% TestCase = atom()
   28:%% Reason = term()
   29:%% @end
   30:%%--------------------------------------------------------------------
   31:all() -> 
   32:    [start_tellstick,
   33:     turn_on].
   34:
   35:%%--------------------------------------------------------------------
   36:%% @spec init_per_suite(Config0) ->
   37:%%     Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   38:%% Config0 = Config1 = [tuple()]
   39:%% Reason = term()
   40:%% @end
   41:%%--------------------------------------------------------------------
   42:init_per_suite(Config) ->
   43:    tellstick_node:init(),
   44:    tellstick_node:start(),
   45:    Config.
   46:
   47:%%--------------------------------------------------------------------
   48:%% @spec end_per_suite(Config0) -> void() | {save_config,Config1}
   49:%% Config0 = Config1 = [tuple()]
   50:%% @end
   51:%%--------------------------------------------------------------------
   52:end_per_suite(_Config) ->
   53:    tellstick_node:stop(),
   54:    ok.
   55:
   56:
   57:%%--------------------------------------------------------------------
   58:%% @spec init_per_testcase(TestCase, Config0) ->
   59:%%               Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   60:%% TestCase = atom()
   61:%% Config0 = Config1 = [tuple()]
   62:%% Reason = term()
   63:%% @end
   64:%%--------------------------------------------------------------------
   65:init_per_testcase(_TestCase, Config) ->
   66:    ct:pal("Testcase: ~p", [_TestCase]),
   67:    DataDir = ?config(data_dir, Config),
   68:    ConfFile = filename:join(DataDir, "tellstick.conf"),
   69:    tellstick_srv:start_link([{debug, true}, 
   70:			      {co_node, tellstick_node:serial()},
   71:			      {config, ConfFile}]),
   72:    Config.
   73:
   74:%%--------------------------------------------------------------------
   75:%% @spec end_per_testcase(TestCase, Config0) ->
   76:%%               void() | {save_config,Config1} | {fail,Reason}
   77:%% TestCase = atom()
   78:%% Config0 = Config1 = [tuple()]
   79:%% Reason = term()
   80:%% @end
   81:%%--------------------------------------------------------------------
   82:end_per_testcase(start_tellstick, _Config) ->
   83:    tellstick_srv:stop(),
   84:    ok;
   85:end_per_testcase(_TestCase, _Config) ->
   86:    ok.
   87:
   88:
   89:%%--------------------------------------------------------------------
   90:%% @spec start_tellstick() -> ok
   91:%% @end
   92:%%--------------------------------------------------------------------
   93:start_tellstick(Config) -> 
   94:    %% Done in init_per_testcase()
   95:    timer:sleep(1000),
   96:    ok.
   97:
   98:%%--------------------------------------------------------------------
   99:%% @spec turn_on() -> ok
  100:%% @end
  101:%%--------------------------------------------------------------------
  102:turn_on(Config) ->
  103:    co_api:notify({xnodeid, serial()}, pdo1_tx, 16#2800, 0, <<1:32/little>>),
  104:    %% How verify ??
  105:    timer:sleep(1000),
  106:    ok.
  107:
  108:%%--------------------------------------------------------------------
  109:%% @spec turn_off() -> ok
  110:%% @end
  111:%%--------------------------------------------------------------------
  112:turn_off(Config) ->
  113:    co_api:notify({xnodeid, serial()}, pdo1_tx, 16#2801, 0, <<1:32/little>>),
  114:    %% How verify ??
  115:    timer:sleep(1000),
  116:    ok.
  117:
  118:%%--------------------------------------------------------------------
  119:%% @spec break(Config) -> ok 
  120:%% @doc 
  121:%% Dummy test case to have a test environment running.
  122:%% Stores Config in ets table.
  123:%% @end
  124:%%--------------------------------------------------------------------
  125:break(Config) ->
  126:    ets:new(config, [set, public, named_table]),
  127:    ets:insert(config, Config),
  128:    test_server:break("Break for test development\n" ++
  129:		     "Get Config by ets:tab2list(config)"),
  130:    ok.
  131:
  132:
  133:serial() ->
  134:    tellstick_node:serial().
