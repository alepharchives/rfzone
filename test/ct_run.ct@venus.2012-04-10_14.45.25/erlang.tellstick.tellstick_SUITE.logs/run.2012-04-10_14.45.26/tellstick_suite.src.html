    9:-module(tellstick_SUITE).
   10:
   11:-compile(export_all).
   12:
   13:-include_lib("common_test/include/ct.hrl").
   14:
   15:%%--------------------------------------------------------------------
   16:%% @spec suite() -> Info
   17:%% Info = [tuple()]
   18:%% @end
   19:%%--------------------------------------------------------------------
   20:suite() ->
   21:    [{timetrap,{seconds,30}}].
   22:
   23:%%--------------------------------------------------------------------
   24:%% @spec all() -> GroupsAndTestCases | {skip,Reason}
   25:%% GroupsAndTestCases = [{group,GroupName} | TestCase]
   26:%% GroupName = atom()
   27:%% TestCase = atom()
   28:%% Reason = term()
   29:%% @end
   30:%%--------------------------------------------------------------------
   31:all() -> 
   32:    [start_tellstick].
   33:
   34:%%--------------------------------------------------------------------
   35:%% @spec init_per_suite(Config0) ->
   36:%%     Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   37:%% Config0 = Config1 = [tuple()]
   38:%% Reason = term()
   39:%% @end
   40:%%--------------------------------------------------------------------
   41:init_per_suite(Config) ->
   42:    tellstick_node:init(),
   43:    Config.
   44:
   45:%%--------------------------------------------------------------------
   46:%% @spec end_per_suite(Config0) -> void() | {save_config,Config1}
   47:%% Config0 = Config1 = [tuple()]
   48:%% @end
   49:%%--------------------------------------------------------------------
   50:end_per_suite(_Config) ->
   51:    tellstick_node:stop(),
   52:    ok.
   53:
   54:
   55:%%--------------------------------------------------------------------
   56:%% @spec init_per_testcase(TestCase, Config0) ->
   57:%%               Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   58:%% TestCase = atom()
   59:%% Config0 = Config1 = [tuple()]
   60:%% Reason = term()
   61:%% @end
   62:%%--------------------------------------------------------------------
   63:init_per_testcase(_TestCase, Config) ->
   64:    Config.
   65:
   66:%%--------------------------------------------------------------------
   67:%% @spec end_per_testcase(TestCase, Config0) ->
   68:%%               void() | {save_config,Config1} | {fail,Reason}
   69:%% TestCase = atom()
   70:%% Config0 = Config1 = [tuple()]
   71:%% Reason = term()
   72:%% @end
   73:%%--------------------------------------------------------------------
   74:end_per_testcase(_TestCase, _Config) ->
   75:    ok.
   76:
   77:
   78:%%--------------------------------------------------------------------
   79:%% @spec TestCase() -> Info
   80:%% Info = [tuple()]
   81:%% @end
   82:%%--------------------------------------------------------------------
   83:start_tellstick(Config) -> 
   84:    DataDir = ?config(data_dir, Config),
   85:    ConfFile = filename:join(DataDir, "tellstick.conf"),
   86:    tellstick_srv:start_link([{debug, true}, 
   87:			      {co_node, tellstick_node:serial()},
   88:			      {config, ConfFile}]),
   89:    
   90:    test_server:break("Break for test development\n"),
   91:
   92:    ok.
   93:
   94:%%--------------------------------------------------------------------
   95:%% @spec break(Config) -> ok 
   96:%% @doc 
   97:%% Dummy test case to have a test environment running.
   98:%% Stores Config in ets table.
   99:%% @end
  100:%%--------------------------------------------------------------------
  101:break(Config) ->
  102:    ets:new(config, [set, public, named_table]),
  103:    ets:insert(config, Config),
  104:    test_server:break("Break for test development\n" ++
  105:		     "Get Config by ets:tab2list(config)"),
  106:    ok.
