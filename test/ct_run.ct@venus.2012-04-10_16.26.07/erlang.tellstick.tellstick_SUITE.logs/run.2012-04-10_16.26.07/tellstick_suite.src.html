   15:%%--------------------------------------------------------------------
   16:%% @spec suite() -> Info
   17:%% Info = [tuple()]
   18:%% @end
   19:%%--------------------------------------------------------------------
   20:suite() ->
   21:    [{timetrap,{seconds,30}}].
   22:
   23:%%--------------------------------------------------------------------
   24:%% @spec all() -> GroupsAndTestCases | {skip,Reason}
   25:%% GroupsAndTestCases = [{group,GroupName} | TestCase]
   26:%% GroupName = atom()
   27:%% TestCase = atom()
   28:%% Reason = term()
   29:%% @end
   30:%%--------------------------------------------------------------------
   31:all() -> 
   32:    [start_tellstick,
   33:     turn_on,
   34:     turn_off].
   35:
   36:%%--------------------------------------------------------------------
   37:%% @spec init_per_suite(Config0) ->
   38:%%     Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   39:%% Config0 = Config1 = [tuple()]
   40:%% Reason = term()
   41:%% @end
   42:%%--------------------------------------------------------------------
   43:init_per_suite(Config) ->
   44:    tellstick_node:init(),
   45:    tellstick_node:start(),
   46:    Config.
   47:
   48:%%--------------------------------------------------------------------
   49:%% @spec end_per_suite(Config0) -> void() | {save_config,Config1}
   50:%% Config0 = Config1 = [tuple()]
   51:%% @end
   52:%%--------------------------------------------------------------------
   53:end_per_suite(_Config) ->
   54:    tellstick_node:stop(),
   55:    ok.
   56:
   57:
   58:%%--------------------------------------------------------------------
   59:%% @spec init_per_testcase(TestCase, Config0) ->
   60:%%               Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   61:%% TestCase = atom()
   62:%% Config0 = Config1 = [tuple()]
   63:%% Reason = term()
   64:%% @end
   65:%%--------------------------------------------------------------------
   66:init_per_testcase(_TestCase, Config) ->
   67:    ct:pal("Testcase: ~p", [_TestCase]),
   68:    ConfFile = filename:join(?config(data_dir, Config), ct:get_config(conf)),
   69:    tellstick_srv:start_link([{debug, true}, 
   70:			      {linked, false},
   71:			      {reset, true},
   72:			      {co_node, tellstick_node:serial()},
   73:			      {config, ConfFile}]),
   74:    Config.
   75:
   76:%%--------------------------------------------------------------------
   77:%% @spec end_per_testcase(TestCase, Config0) ->
   78:%%               void() | {save_config,Config1} | {fail,Reason}
   79:%% TestCase = atom()
   80:%% Config0 = Config1 = [tuple()]
   81:%% Reason = term()
   82:%% @end
   83:%%--------------------------------------------------------------------
   84:end_per_testcase(_TestCase, _Config) ->
   85:    tellstick_srv:stop(),
   86:    ok.
   87:
   88:
   89:%%--------------------------------------------------------------------
   90:%% @spec start_tellstick() -> ok
   91:%% @end
   92:%%--------------------------------------------------------------------
   93:start_tellstick(_Config) -> 
   94:    %% Done in init_per_testcase()
   95:    timer:sleep(1000),
   96:    ok.

  128:%%--------------------------------------------------------------------
  129:%% @spec break(Config) -> ok 
  130:%% @doc 
  131:%% Dummy test case to have a test environment running.
  132:%% Stores Config in ets table.
  133:%% @end
  134:%%--------------------------------------------------------------------
  135:break(Config) ->
  136:    ets:new(config, [set, public, named_table]),
