    9:-module(tellstick_SUITE).
   10:
   11:-compile(export_all).
   12:
   13:-include_lib("common_test/include/ct.hrl").
   14:
   15:%%--------------------------------------------------------------------
   16:%% @spec suite() -> Info
   17:%% Info = [tuple()]
   18:%% @end
   19:%%--------------------------------------------------------------------
   20:suite() ->
   21:    [{timetrap,{seconds,30}}].
   22:
   23:%%--------------------------------------------------------------------
   24:%% @spec all() -> GroupsAndTestCases | {skip,Reason}
   25:%% GroupsAndTestCases = [{group,GroupName} | TestCase]
   26:%% GroupName = atom()
   27:%% TestCase = atom()
   28:%% Reason = term()
   29:%% @end
   30:%%--------------------------------------------------------------------
   31:all() -> 
   32:    [start_tellstick].
   33:
   34:%%--------------------------------------------------------------------
   35:%% @spec init_per_suite(Config0) ->
   36:%%     Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   37:%% Config0 = Config1 = [tuple()]
   38:%% Reason = term()
   39:%% @end
   40:%%--------------------------------------------------------------------
   41:init_per_suite(Config) ->
   42:    tellstick_node:init(),
   43:    tellstick_node:start(),
   44:    Config.
   45:
   46:%%--------------------------------------------------------------------
   47:%% @spec end_per_suite(Config0) -> void() | {save_config,Config1}
   48:%% Config0 = Config1 = [tuple()]
   49:%% @end
   50:%%--------------------------------------------------------------------
   51:end_per_suite(_Config) ->
   52:    tellstick_node:stop(),
   53:    ok.
   54:
   55:
   56:%%--------------------------------------------------------------------
   57:%% @spec init_per_testcase(TestCase, Config0) ->
   58:%%               Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   59:%% TestCase = atom()
   60:%% Config0 = Config1 = [tuple()]
   61:%% Reason = term()
   62:%% @end
   63:%%--------------------------------------------------------------------
   64:init_per_testcase(_TestCase, Config) ->
   65:    Config.
   66:
   67:%%--------------------------------------------------------------------
   68:%% @spec end_per_testcase(TestCase, Config0) ->
   69:%%               void() | {save_config,Config1} | {fail,Reason}
   70:%% TestCase = atom()
   71:%% Config0 = Config1 = [tuple()]
   72:%% Reason = term()
   73:%% @end
   74:%%--------------------------------------------------------------------
   75:end_per_testcase(_TestCase, _Config) ->
   76:    ok.
   77:
   78:
   79:%%--------------------------------------------------------------------
   80:%% @spec TestCase() -> Info
   81:%% Info = [tuple()]
   82:%% @end
   83:%%--------------------------------------------------------------------
   84:start_tellstick(Config) -> 
   85:    DataDir = ?config(data_dir, Config),
   86:    ConfFile = filename:join(DataDir, "tellstick.conf"),
   87:    tellstick_srv:start_link([{debug, true}, 
   88:			      {co_node, tellstick_node:serial()},
   89:			      {config, ConfFile}]),
   90:    
   91:    test_server:break("Break for test development\n"),
   92:
   93:    ok.
   94:
   95:%%--------------------------------------------------------------------
   96:%% @spec break(Config) -> ok 
   97:%% @doc 
   98:%% Dummy test case to have a test environment running.
   99:%% Stores Config in ets table.
  100:%% @end
  101:%%--------------------------------------------------------------------
  102:break(Config) ->
  103:    ets:new(config, [set, public, named_table]),
  104:    ets:insert(config, Config),
  105:    test_server:break("Break for test development\n" ++
  106:		     "Get Config by ets:tab2list(config)"),
  107:    ok.
