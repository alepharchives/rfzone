    9:-module(tellstick_SUITE).
   10:
   11:-compile(export_all).
   12:
   13:-include_lib("common_test/include/ct.hrl").
   14:
   15:%%--------------------------------------------------------------------
   16:%% @spec suite() -> Info
   17:%% Info = [tuple()]
   18:%% @end
   19:%%--------------------------------------------------------------------
   20:suite() ->
   21:    [{timetrap,{seconds,30}}].
   22:
   23:%%--------------------------------------------------------------------
   24:%% @spec all() -> GroupsAndTestCases | {skip,Reason}
   25:%% GroupsAndTestCases = [{group,GroupName} | TestCase]
   26:%% GroupName = atom()
   27:%% TestCase = atom()
   28:%% Reason = term()
   29:%% @end
   30:%%--------------------------------------------------------------------
   31:all() -> 
   32:    [start_tellstick,
   33:     turn_on,
   34:     turn_off].
   35:
   36:%%--------------------------------------------------------------------
   37:%% @spec init_per_suite(Config0) ->
   38:%%     Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   39:%% Config0 = Config1 = [tuple()]
   40:%% Reason = term()
   41:%% @end
   42:%%--------------------------------------------------------------------
   43:init_per_suite(Config) ->
   44:    tellstick_node:init(),
   45:    tellstick_node:start(),
   46:    Config.
   47:
   48:%%--------------------------------------------------------------------
   49:%% @spec end_per_suite(Config0) -> void() | {save_config,Config1}
   50:%% Config0 = Config1 = [tuple()]
   51:%% @end
   52:%%--------------------------------------------------------------------
   53:end_per_suite(_Config) ->
   54:    tellstick_node:stop(),
   55:    ok.
   56:
   57:
   58:%%--------------------------------------------------------------------
   59:%% @spec init_per_testcase(TestCase, Config0) ->
   60:%%               Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   61:%% TestCase = atom()
   62:%% Config0 = Config1 = [tuple()]
   63:%% Reason = term()
   64:%% @end
   65:%%--------------------------------------------------------------------
   66:init_per_testcase(_TestCase, Config) ->
   67:    ct:pal("Testcase: ~p", [_TestCase]),
   68:    DataDir = ?config(data_dir, Config),
   69:    ConfFile = filename:join(DataDir, "tellstick.conf"),
   70:    tellstick_srv:start_link([{debug, true}, 
   71:			      {linked, false},
   72:			      {co_node, tellstick_node:serial()},
   73:			      {config, ConfFile}]),
   74:    Config.
   75:
   76:%%--------------------------------------------------------------------
   77:%% @spec end_per_testcase(TestCase, Config0) ->
   78:%%               void() | {save_config,Config1} | {fail,Reason}
   79:%% TestCase = atom()
   80:%% Config0 = Config1 = [tuple()]
   81:%% Reason = term()
   82:%% @end
   83:%%--------------------------------------------------------------------
   84:end_per_testcase(_TestCase, _Config) ->
   85:    tellstick_srv:stop(),
   86:    ok.
   87:
   88:
   89:%%--------------------------------------------------------------------
   90:%% @spec start_tellstick() -> ok
   91:%% @end
   92:%%--------------------------------------------------------------------
   93:start_tellstick(Config) -> 
   94:    %% Done in init_per_testcase()
   95:    timer:sleep(1000),
   96:    ok.
   97:
   98:%%--------------------------------------------------------------------
   99:%% @spec turn_on() -> ok
  100:%% @end
  101:%%--------------------------------------------------------------------
  102:turn_on(Config) ->
  103:    co_api:notify({xnodeid, serial()}, pdo1_tx, 16#6000, 9, <<1:32/little>>),
  104:    %% How verify ??
  105:    timer:sleep(1000),
  106:    ok.
  107:
  108:%%--------------------------------------------------------------------
  109:%% @spec turn_off() -> ok
  110:%% @end
  111:%%--------------------------------------------------------------------
  112:turn_off(Config) ->
  113:    co_api:notify({xnodeid, serial()}, pdo1_tx, 16#6000, 9, <<0:32/little>>),
  114:    %% How verify ??
  115:    timer:sleep(1000),
  116:    ok.
  117:
  118:%%--------------------------------------------------------------------
  119:%% @spec break(Config) -> ok 
  120:%% @doc 
  121:%% Dummy test case to have a test environment running.
  122:%% Stores Config in ets table.
  123:%% @end
  124:%%--------------------------------------------------------------------
  125:break(Config) ->
  126:    ets:new(config, [set, public, named_table]),
  127:    ets:insert(config, Config),
  128:    test_server:break("Break for test development\n" ++
  129:		     "Get Config by ets:tab2list(config)"),
  130:    ok.
  131:
  132:
  133:serial() ->
  134:    tellstick_node:serial().
