    9:-module(tellstick_SUITE).
   10:
   11:-compile(export_all).
   12:
   13:-include_lib("common_test/include/ct.hrl").
   14:
   15:%%--------------------------------------------------------------------
   16:%% @spec suite() -> Info
   17:%% Info = [tuple()]
   18:%% @end
   19:%%--------------------------------------------------------------------
   20:suite() ->
   21:    [{timetrap,{seconds,30}}].
   22:
   23:%%--------------------------------------------------------------------
   24:%% @spec all() -> GroupsAndTestCases | {skip,Reason}
   25:%% GroupsAndTestCases = [{group,GroupName} | TestCase]
   26:%% GroupName = atom()
   27:%% TestCase = atom()
   28:%% Reason = term()
   29:%% @end
   30:%%--------------------------------------------------------------------
   31:all() -> 
   32:    [start_tellstick,
   33:     turn_on,
   34:     turn_off].
   35:
   36:%%--------------------------------------------------------------------
   37:%% @spec init_per_suite(Config0) ->
   38:%%     Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   39:%% Config0 = Config1 = [tuple()]
   40:%% Reason = term()
   41:%% @end
   42:%%--------------------------------------------------------------------
   43:init_per_suite(Config) ->
   44:    tellstick_node:init(),
   45:    tellstick_node:start(),
   46:    Config.
   47:
   48:%%--------------------------------------------------------------------
   49:%% @spec end_per_suite(Config0) -> void() | {save_config,Config1}
   50:%% Config0 = Config1 = [tuple()]
   51:%% @end
   52:%%--------------------------------------------------------------------
   53:end_per_suite(_Config) ->
   54:    tellstick_node:stop(),
   55:    ok.
   56:
   57:
   58:%%--------------------------------------------------------------------
   59:%% @spec init_per_testcase(TestCase, Config0) ->
   60:%%               Config1 | {skip,Reason} | {skip_and_save,Reason,Config1}
   61:%% TestCase = atom()
   62:%% Config0 = Config1 = [tuple()]
   63:%% Reason = term()
   64:%% @end
   65:%%--------------------------------------------------------------------
   66:init_per_testcase(_TestCase, Config) ->
   67:    ct:pal("Testcase: ~p", [_TestCase]),
   68:    ConfFile = filename:join(?config(data_dir, Config), ct:get_config(conf)),
   69:    tellstick_srv:start_link([{debug, true}, 
   70:			      {linked, false},
   71:			      {co_node, tellstick_node:serial()},
   72:			      {config, ConfFile}]),
   73:    Config.
   74:
   75:%%--------------------------------------------------------------------
   76:%% @spec end_per_testcase(TestCase, Config0) ->
   77:%%               void() | {save_config,Config1} | {fail,Reason}
   78:%% TestCase = atom()
   79:%% Config0 = Config1 = [tuple()]
   80:%% Reason = term()
   81:%% @end
   82:%%--------------------------------------------------------------------
   83:end_per_testcase(_TestCase, _Config) ->
   84:    tellstick_srv:stop(),
   85:    ok.
   86:
   87:
   88:%%--------------------------------------------------------------------
   89:%% @spec start_tellstick() -> ok
   90:%% @end
   91:%%--------------------------------------------------------------------
   92:start_tellstick(_Config) -> 
   93:    %% Done in init_per_testcase()
   94:    timer:sleep(1000),
   95:    ok.
   96:
   97:%%--------------------------------------------------------------------
   98:%% @spec turn_on() -> ok
   99:%% @end
  100:%%--------------------------------------------------------------------
  101:turn_on(_Config) ->
  102:    %% Springback requires on and off
  103:    co_api:notify({xnodeid, ct:get_config(source_node)}, 
  104:		  pdo1_tx, 16#6000, 9, <<1:32/little>>),
  105:    timer:sleep(100),
  106:    co_api:notify({xnodeid, ct:get_config(source_node)}, 
  107:		  pdo1_tx, 16#6000, 9, <<0:32/little>>),
  108:    %% How verify ??
  109:    timer:sleep(1000),
  110:    ok.
  111:
  112:%%--------------------------------------------------------------------
  113:%% @spec turn_off() -> ok
  114:%% @end
  115:%%--------------------------------------------------------------------
  116:turn_off(_Config) ->
  117:    %% Springback requires on and off
  118:    co_api:notify({xnodeid, ct:get_config(source_node)}, 
  119:		  pdo1_tx, 16#6000, 9, <<1:32/little>>),
  120:    timer:sleep(100),
  121:    co_api:notify({xnodeid, ct:get_config(source_node)}, 
  122:		  pdo1_tx, 16#6000, 9, <<0:32/little>>),
  123:    %% How verify ??
  124:    timer:sleep(1000),
  125:    ok.
  126:
  127:%%--------------------------------------------------------------------
  128:%% @spec break(Config) -> ok 
  129:%% @doc 
  130:%% Dummy test case to have a test environment running.
  131:%% Stores Config in ets table.
  132:%% @end
  133:%%--------------------------------------------------------------------
  134:break(Config) ->
  135:    ets:new(config, [set, public, named_table]),
  136:    ets:insert(config, Config),
  137:    test_server:break("Break for test development\n" ++
  138:		     "Get Config by ets:tab2list(config)"),
  139:    ok.
  140:
  141:
  142:serial() ->
  143:    tellstick_node:serial().
